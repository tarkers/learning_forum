tmp1 = people.manager_information.find( board_ID = p_board_ID )
if tmp1[0]['password_private'] == p_password
    people.manager_information.updateOne(where{board_ID = p_board_ID},$set{p_type = p_include})
    class = tmp1[0]['class']
    switch(p_type)
        case title
            board.all_board_number.updateOne(where{board_ID = p_board_ID},$set{title = p_include})
            data.class_board.updateOne(where{board_ID = p_board_ID},$set{title = p_include})
        case tab
            data.class_board.updateOne(where{board_ID = p_board_ID},$set{tab = p_include})
        case include
